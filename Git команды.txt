git init - следить за папкой в которой работаете
git config --local(или global) user.name(потом "email") "Имя" почта - настройка имени папки
git add -A - при создании новых файлов команда для коммита и создания контрольной точки в дальнейшем
git commit -a -m"описание контрольной точки" - создание контрольной точки
git log - просмотреть коммиты(сохранения)
